## 📌 技术概述

### 项目技术特点
- **视频互动**: 大量视频素材的加载、播放、无缝切换
- **分支剧情**: 复杂的剧情分支逻辑和状态管理
- **Web平台**: 浏览器端运行，需要考虑兼容性和性能
- **第一人称**: 沉浸式体验，需要流畅的界面交互

### 技术挑战（基于市场调研）
根据市场调研，用户主要痛点集中在：
1. **视频加载慢** (38%用户反馈) - 需重点优化
2. **剧情分支差异小** (45%用户反馈) - 需设计灵活的剧情系统
3. **重复游玩枯燥** (42%用户反馈) - 需增加隐藏内容和随机元素

### 技术目标

| 指标 | 目标值 | 依据 |
|------|--------|------|
| 首屏加载时间 | < 5秒 | 单机游戏包体较大 |
| 视频起播时间 | < 1秒 | 本地资源无需网络 |
| 视频切换延迟 | < 300ms | 本地读取更快 |
| 内存占用峰值 | < 1GB | 单机游戏可占用更多 |
| 移动端流畅度 | ≥ 30fps | 基本流畅标准 |

### 单机模式优势

| 优势 | 说明 |
|------|------|
| **零服务器成本** | 无需购买/租用服务器、数据库、CDN等云服务 |
| **零运维成本** | 无需运维人员、无需监控告警 |
| **零网络依赖** | 玩家可离线游玩，无网络延迟 |
| **开发门槛低** | 纯前端开发，无需后端知识 |
| **部署简单** | 静态文件部署到任意Web服务器或本地运行 |

### 核心模块设计

#### 1. 视频管理器 (VideoManager)

```gdscript
## 视频管理器 - 负责视频加载、播放、缓存（单机版）
class_name VideoManager extends Node

## 信号定义
signal video_loaded(video_id: String)
signal video_error(video_id: String, error: String)
signal video_progress(video_id: String, progress: float)
signal video_completed(video_id: String)

## 配置
const MAX_CACHE_SIZE: int = 5           # 最大缓存视频数
const PRELOAD_COUNT: int = 2            # 预加载后续视频数
const VIDEO_BASE_PATH: String = "res://assets/videos/"  # 视频资源路径

## 状态
var _video_pool: Dictionary = {}        # 视频缓存池
var _current_video: String = ""         # 当前播放视频
var _is_playing: bool = false           # 播放状态
var _video_player: VideoStreamPlayer    # 视频播放器

## 核心方法
func load_video(video_id: String) -> void:
    if _video_pool.has(video_id):
        video_loaded.emit(video_id)
        return

    var video_path = _get_video_path(video_id)
    if not ResourceLoader.exists(video_path):
        video_error.emit(video_id, "Video file not found: " + video_path)
        return

    var video_stream = load(video_path)
    if video_stream == null:
        video_error.emit(video_id, "Failed to load video")
        return

    _video_pool[video_id] = video_stream
    video_loaded.emit(video_id)

func play_video(video_id: String) -> void:
    if not _video_pool.has(video_id):
        load_video(video_id)
        await video_loaded

    _current_video = video_id
    _is_playing = true
    _video_player.stream = _video_pool[video_id]
    _video_player.play()

func preload_next(node_id: String) -> void:
    # 根据当前节点预测下一个节点，预加载视频
    var next_nodes = StoryEngine.get_possible_next_nodes(node_id)
    for i in range(min(next_nodes.size(), PRELOAD_COUNT)):
        var video_id = StoryEngine.get_video_id(next_nodes[i])
        load_video(video_id)

func clear_cache() -> void:
    for key in _video_pool.keys():
        _video_pool.erase(key)
    _video_pool.clear()

func _get_video_path(video_id: String) -> String:
    return VIDEO_BASE_PATH + video_id + ".ogv"
```

#### 2. 剧情引擎 (StoryEngine)

```gdscript
## 剧情引擎 - 负责剧情流程控制
class_name StoryEngine extends Node

## 信号定义
signal story_started(chapter_id: String)
signal node_changed(node_id: String)
signal choices_available(choices: Array)
signal choice_made(choice_id: String)
signal story_ended(ending_id: String)

## 状态
var _current_chapter: String = ""
var _current_node: String = ""
var _story_data: Dictionary = {}
var _visited_nodes: Array = []

## 核心方法
func start_chapter(chapter_id: String) -> void:
    _current_chapter = chapter_id
    _load_chapter_data(chapter_id)
    _jump_to_node(_story_data["start_node"])
    story_started.emit(chapter_id)

func make_choice(choice_index: int) -> void:
    var choices = get_available_choices()
    if choice_index >= choices.size():
        return

    var choice = choices[choice_index]

    # 检查条件
    if not _check_conditions(choice.get("conditions", [])):
        return

    # 执行效果
    _apply_effects(choice.get("effects", []))

    # 记录历史
    _visited_nodes.append(_current_node)
    GameStateManager.record_choice(choice["choice_id"])

    # 跳转下一节点
    _jump_to_node(choice["next_node"])
    choice_made.emit(choice["choice_id"])

func get_available_choices() -> Array:
    var node = get_current_node()
    var available = []

    for choice in node.get("choices", []):
        if _check_conditions(choice.get("conditions", [])):
            available.append(choice)

    return available

func _check_conditions(conditions: Array) -> bool:
    for cond in conditions:
        match cond["type"]:
            "flag":
                if GameStateManager.get_flag(cond["key"]) != cond["value"]:
                    return false
            "variable":
                var val = GameStateManager.get_variable(cond["key"])
                if not _compare_value(val, cond["operator"], cond["value"]):
                    return false
            "item":
                if not GameStateManager.has_item(cond["item_id"]):
                    return false
    return true

func _apply_effects(effects: Array) -> void:
    for effect in effects:
        match effect["type"]:
            "set_flag":
                GameStateManager.set_flag(effect["key"], effect["value"])
            "set_variable":
                GameStateManager.set_variable(effect["key"], effect["value"])
            "add_item":
                GameStateManager.add_item(effect["item_id"])
            "remove_item":
                GameStateManager.remove_item(effect["item_id"])
            "trigger_event":
                EventManager.trigger(effect["event_id"])
```

#### 3. 状态管理器 (GameStateManager)

```gdscript
## 状态管理器 - 负责游戏全局状态
class_name GameStateManager extends Node

## 状态数据
var _flags: Dictionary = {}         # 布尔标记
var _variables: Dictionary = {}     # 数值变量
var _items: Array = []              # 物品列表
var _relationships: Dictionary = {} # NPC好感度
var _choice_history: Array = []     # 选择历史

## 常量定义
const INITIAL_TRUST: int = 50       # 初始好感度

## 核心方法
func set_flag(key: String, value: Variant) -> void:
    _flags[key] = value
    EventManager.emit("flag_changed", {"key": key, "value": value})

func get_flag(key: String, default: Variant = false) -> Variant:
    return _flags.get(key, default)

func set_variable(key: String, value: Variant) -> void:
    _variables[key] = value

func get_variable(key: String, default: Variant = 0) -> Variant:
    return _variables.get(key, default)

func modify_variable(key: String, delta: float) -> void:
    var current = get_variable(key, 0)
    set_variable(key, current + delta)

func add_item(item_id: String, count: int = 1) -> void:
    for i in count:
        _items.append(item_id)

func remove_item(item_id: String, count: int = 1) -> bool:
    var removed = 0
    for i in range(_items.size() - 1, -1, -1):
        if _items[i] == item_id and removed < count:
            _items.remove_at(i)
            removed += 1
    return removed == count

func has_item(item_id: String) -> bool:
    return item_id in _items

func get_relationship(npc_id: String) -> int:
    return _relationships.get(npc_id, INITIAL_TRUST)

func modify_relationship(npc_id: String, delta: int) -> void:
    var current = get_relationship(npc_id)
    var new_value = clamp(current + delta, 0, 100)
    _relationships[npc_id] = new_value

func record_choice(choice_id: String) -> void:
    _choice_history.append({
        "choice_id": choice_id,
        "timestamp": Time.get_unix_time_from_system()
    })
```

#### 4. 存档管理器 (SaveManager)

```gdscript
## 存档管理器 - 负责存档读写（单机版）
class_name SaveManager extends Node

## 配置
const SAVE_KEY_PREFIX: String = "lost_choices_save_"
const MAX_SLOTS: int = 5
const AUTO_SAVE_INTERVAL: float = 60.0  # 自动存档间隔(秒)
const SAVE_VERSION: int = 1             # 存档版本号

## 信号
signal save_completed(slot: int, success: bool)
signal load_completed(slot: int, success: bool)

## 核心方法
func save_game(slot: int) -> bool:
    if slot < 0 or slot >= MAX_SLOTS:
        return false

    var save_data = {
        "version": SAVE_VERSION,
        "timestamp": Time.get_unix_time_from_system(),
        "chapter": StoryEngine.get_current_chapter(),
        "node": StoryEngine.get_current_node(),
        "flags": GameStateManager.get_all_flags(),
        "variables": GameStateManager.get_all_variables(),
        "items": GameStateManager.get_all_items(),
        "relationships": GameStateManager.get_all_relationships(),
        "history": GameStateManager.get_choice_history(),
        "achievements": AchievementManager.get_unlocked(),
        "play_time": GameManager.get_play_time(),
        "screen_shot": _capture_thumbnail()
    }

    var json_string = JSON.stringify(save_data)
    var encrypted_data = SaveEncryption.encrypt(json_string)
    var key = SAVE_KEY_PREFIX + str(slot)

    # Web端使用IndexedDB存储
    if OS.has_feature("web"):
        _save_to_indexed_db(slot, encrypted_data)
    else:
        # 桌面端使用文件存储
        _save_to_file(slot, encrypted_data)

    save_completed.emit(slot, true)
    return true

func load_game(slot: int) -> bool:
    var key = SAVE_KEY_PREFIX + str(slot)

    # Web端从IndexedDB读取
    if OS.has_feature("web"):
        var data = await _load_from_indexed_db(slot)
        if data.is_empty():
            load_completed.emit(slot, false)
            return false
        _apply_save_data(data)
    else:
        # 桌面端从文件读取
        var data = _load_from_file(slot)
        if data.is_empty():
            load_completed.emit(slot, false)
            return false
        _apply_save_data(data)

    load_completed.emit(slot, true)
    return true

func get_save_info(slot: int) -> Dictionary:
    var key = SAVE_KEY_PREFIX + str(slot)
    var info = {}

    if OS.has_feature("web"):
        info = _get_indexed_db_save_info(slot)
    else:
        info = _get_file_save_info(slot)

    return info

func delete_save(slot: int) -> bool:
    var key = SAVE_KEY_PREFIX + str(slot)

    if OS.has_feature("web"):
        _delete_from_indexed_db(slot)
    else:
        _delete_file(slot)

    return true

func auto_save() -> void:
    save_game(0)  # 槽位0用于自动存档

## IndexedDB 操作（Web端）
func _save_to_indexed_db(slot: int, data: String) -> void:
    JavaScriptBridge.eval("""
        (function() {
            var request = indexedDB.open('LostChoicesSaves', 1);
            request.onupgradeneeded = function(e) {
                var db = e.target.result;
                if (!db.objectStoreNames.contains('saves')) {
                    db.createObjectStore('saves', {keyPath: 'slot'});
                }
            };
            request.onsuccess = function(e) {
                var db = e.target.result;
                var tx = db.transaction('saves', 'readwrite');
                var store = tx.objectStore('saves');
                store.put({slot: %d, data: '%s', timestamp: Date.now()});
            };
        })();
    """ % [slot, data])

func _load_from_indexed_db(slot: int) -> String:
    var result = JavaScriptBridge.eval("""
        (function() {
            return new Promise(function(resolve) {
                var request = indexedDB.open('LostChoicesSaves', 1);
                request.onsuccess = function(e) {
                    var db = e.target.result;
                    var tx = db.transaction('saves', 'readonly');
                    var store = tx.objectStore('saves');
                    var getRequest = store.get(%d);
                    getRequest.onsuccess = function(e) {
                        resolve(e.target.result ? e.target.result.data : '');
                    };
                    getRequest.onerror = function() {
                        resolve('');
                    };
                };
                request.onerror = function() {
                    resolve('');
                };
            });
        })();
    """ % [slot])
    return result if result else ""

## 文件操作（桌面端）
func _save_to_file(slot: int, data: String) -> void:
    var save_dir = OS.get_user_data_dir() + "/saves"
    DirAccess.make_dir_recursive_absolute(save_dir)
    var file = FileAccess.open(save_dir + "/slot_%d.save" % slot, FileAccess.WRITE)
    file.store_string(data)
    file.close()

func _load_from_file(slot: int) -> String:
    var save_path = OS.get_user_data_dir() + "/saves/slot_%d.save" % slot
    if not FileAccess.file_exists(save_path):
        return ""
    var file = FileAccess.open(save_path, FileAccess.READ)
    var data = file.get_as_text()
    file.close()
    return data

func _apply_save_data(encrypted_data: String) -> void:
    var json_string = SaveEncryption.decrypt(encrypted_data)
    var save_data = JSON.parse_string(json_string)

    GameStateManager.load_state(save_data)
    StoryEngine.jump_to_node(save_data["node"])
    AchievementManager.load_achievements(save_data.get("achievements", []))
    GameManager.set_play_time(save_data.get("play_time", 0))

func _capture_thumbnail() -> String:
    # 捕获当前画面作为缩略图
    var viewport = get_viewport()
    var texture = viewport.get_texture()
    var image = texture.get_image()
    image.resize(320, 180)
    return Marshalls.raw_to_base64(image.save_png_to_buffer())
```

```gdscript
## 资源管理器 - 负责本地资源的加载和管理
class_name ResourceManager extends Node

## 资源路径配置
const PATHS: Dictionary = {
    "videos": "res://assets/videos/",
    "audio_bgm": "res://assets/audio/bgm/",
    "audio_sfx": "res://assets/audio/sfx/",
    "images": "res://assets/images/",
    "data": "res://data/"
}

## 资源缓存
var _resource_cache: Dictionary = {}

## 加载资源
func load_resource(path: String, cache: bool = true) -> Resource:
    if cache and _resource_cache.has(path):
        return _resource_cache[path]

    if not ResourceLoader.exists(path):
        push_error("Resource not found: " + path)
        return null

    var resource = load(path)
    if cache and resource != null:
        _resource_cache[path] = resource

    return resource

## 预加载资源
func preload_resources(paths: Array) -> void:
    for path in paths:
        load_resource(path, true)

## 清理缓存
func clear_cache() -> void:
    _resource_cache.clear()

## 获取资源列表
func list_resources(type: String) -> Array:
    var path = PATHS.get(type, "")
    if path.is_empty():
        return []

    var dir = DirAccess.open(path)
    if dir == null:
        return []

    var files = []
    dir.list_dir_begin()
    var file_name = dir.get_next()
    while file_name != "":
        if not dir.current_is_dir():
            files.append(path + file_name)
        file_name = dir.get_next()

    return files
```

### 存档加密实现

```gdscript
## 存档加密工具
class_name SaveEncryption

## 加密密钥（实际项目中应从配置读取，不要硬编码）
const KEY_SALT: String = "lost_choices_2026"

## 生成密钥
static func _generate_key() -> PackedByteArray:
    var key = KEY_SALT.sha256_text()
    return key.to_utf8_buffer()

## 加密数据
static func encrypt(data: String) -> String:
    if data.is_empty():
        return ""

    var key = _generate_key()
    var bytes = data.to_utf8_buffer()

    # 简单的XOR加密（生产环境建议使用AES）
    for i in range(bytes.size()):
        bytes[i] = bytes[i] ^ key[i % key.size()]

    return Marshalls.raw_to_base64(bytes)

## 解密数据
static func decrypt(encrypted_data: String) -> String:
    if encrypted_data.is_empty():
        return ""

    var key = _generate_key()
    var bytes = Marshalls.base64_to_raw(encrypted_data)

    # XOR解密
    for i in range(bytes.size()):
        bytes[i] = bytes[i] ^ key[i % key.size()]

    return bytes.get_string_from_utf8()

## 计算数据Hash
static func calculate_hash(data: String) -> String:
    return data.sha256_text()

## 验证数据完整性
static func verify_integrity(data: String, hash: String) -> bool:
    return calculate_hash(data) == hash
```

```gdscript
## 本地统计管理器（无需云端）
class_name StatsManager extends Node

## 统计数据
var _stats: Dictionary = {

        "chapter_complete":

            if value not in _stats["endings_unlocked"]:
                _stats["endings_unlocked"].append(value)
        "choice_made":
            _stats["choices_made"] += 1
        "video_watched":
            _stats["videos_watched"] += 1

        "chapters_completed": _stats["chapters_completed"].size(),
        "endings_unlocked": _stats["endings_unlocked"].size(),
        "total_play_time_hours": _stats["total_play_time"] / 3600.0
    }

## 保存统计
static func _save_stats() -> void:
    var save_path = _get_stats_path()
    var file = FileAccess.open(save_path, FileAccess.WRITE)
    file.store_string(JSON.stringify(_stats))
    file.close()

## 加载统计
static func load_stats() -> void:
    var save_path = _get_stats_path()
    if FileAccess.file_exists(save_path):
        var file = FileAccess.open(save_path, FileAccess.READ)
        var data = JSON.parse_string(file.get_as_text())
        if data != null:
            _stats = data
        file.close()

static func _get_stats_path() -> String:
    if OS.has_feature("web"):
        # Web端使用localStorage
        return "user://stats.json"
    else:
```

### 单机模式限制

| 限制 | 说明 | 解决方案 |
|------|------|----------|
| 包体较大 | 视频资源打包在内 | 采用分包加载、压缩优化 |
| 无云存档 | 存档仅存本地 | 提供导出/导入功能 |
| 无实时统计 | 无法收集用户数据 | 本地统计、用户主动反馈 |
| 无在线功能 | 无法多人互动 | 设计为纯单机体验 |
