## 📌 技术概述

### 项目技术特点
- **视频互动**: 大量视频素材的加载、播放、无缝切换
- **分支剧情**: 复杂的剧情分支逻辑和状态管理
- **Web平台**: 浏览器端运行，需要考虑兼容性和性能
- **第一人称**: 沉浸式体验，需要流畅的界面交互

### 技术挑战（基于市场调研）
根据市场调研，用户主要痛点集中在：
1. **视频加载慢** (38%用户反馈) - 需重点优化
2. **剧情分支差异小** (45%用户反馈) - 需设计灵活的剧情系统
3. **重复游玩枯燥** (42%用户反馈) - 需增加隐藏内容和随机元素

### 技术目标

| 指标 | 目标值 | 依据 |
|------|--------|------|
| 首屏加载时间 | < 3秒 | 用户流失临界点 |
| 视频起播时间 | < 2秒 | 用户可接受等待时间 |
| 视频切换延迟 | < 500ms | 保证沉浸感 |
| 内存占用峰值 | < 500MB | 浏览器安全阈值 |
| 移动端流畅度 | ≥ 30fps | 基本流畅标准 |

```gdscript
func load_video(video_id: String, quality: String = "") -> void:
    var target_quality = quality if quality else _current_quality

    if _video_pool.has(cache_key):
        video_loaded.emit(video_id)
        return

    var url = _get_video_url(video_id, target_quality)
    _load_from_cdn(url, cache_key)

func play_video(video_id: String) -> void:
    _current_video = video_id
    _is_playing = true
    # 播放逻辑...

func preload_next(node_id: String) -> void:
    # 根据当前节点预测下一个节点，预加载视频
    var next_nodes = StoryEngine.get_possible_next_nodes(node_id)
    for i in range(min(next_nodes.size(), PRELOAD_COUNT)):
        var video_id = StoryEngine.get_video_id(next_nodes[i])
        load_video(video_id)

func clear_cache() -> void:
    for key in _video_pool.keys():
        _video_pool[key].free()
```

#### 2. 剧情引擎 (StoryEngine)

```gdscript
## 剧情引擎 - 负责剧情流程控制
class_name StoryEngine extends Node

## 信号定义
signal story_started(chapter_id: String)
signal node_changed(node_id: String)
signal choices_available(choices: Array)
signal choice_made(choice_id: String)
signal story_ended(ending_id: String)

## 状态
var _current_chapter: String = ""
var _current_node: String = ""
var _story_data: Dictionary = {}
var _visited_nodes: Array = []

## 核心方法
func start_chapter(chapter_id: String) -> void:
    _current_chapter = chapter_id
    _load_chapter_data(chapter_id)
    _jump_to_node(_story_data["start_node"])
    story_started.emit(chapter_id)

func make_choice(choice_index: int) -> void:
    var choices = get_available_choices()
    if choice_index >= choices.size():
        return

    var choice = choices[choice_index]

    # 检查条件
    if not _check_conditions(choice.get("conditions", [])):
        return

    # 执行效果
    _apply_effects(choice.get("effects", []))

    # 记录历史
    _visited_nodes.append(_current_node)
    GameStateManager.record_choice(choice["choice_id"])

    # 跳转下一节点
    _jump_to_node(choice["next_node"])
    choice_made.emit(choice["choice_id"])

func get_available_choices() -> Array:
    var node = get_current_node()
    var available = []

    for choice in node.get("choices", []):
        if _check_conditions(choice.get("conditions", [])):
            available.append(choice)

    return available

func _check_conditions(conditions: Array) -> bool:
    for cond in conditions:
        match cond["type"]:
            "flag":
                if GameStateManager.get_flag(cond["key"]) != cond["value"]:
                    return false
            "variable":
                var val = GameStateManager.get_variable(cond["key"])
                if not _compare_value(val, cond["operator"], cond["value"]):
                    return false
            "item":
                if not GameStateManager.has_item(cond["item_id"]):
                    return false
    return true

func _apply_effects(effects: Array) -> void:
    for effect in effects:
        match effect["type"]:
            "set_flag":
                GameStateManager.set_flag(effect["key"], effect["value"])
            "set_variable":
                GameStateManager.set_variable(effect["key"], effect["value"])
            "add_item":
                GameStateManager.add_item(effect["item_id"])
            "remove_item":
                GameStateManager.remove_item(effect["item_id"])
            "trigger_event":
                EventManager.trigger(effect["event_id"])
```

#### 3. 状态管理器 (GameStateManager)

```gdscript
## 状态管理器 - 负责游戏全局状态
class_name GameStateManager extends Node

## 状态数据
var _flags: Dictionary = {}         # 布尔标记
var _variables: Dictionary = {}     # 数值变量
var _items: Array = []              # 物品列表
var _relationships: Dictionary = {} # NPC好感度
var _choice_history: Array = []     # 选择历史

## 常量定义
const INITIAL_TRUST: int = 50       # 初始好感度

## 核心方法
func set_flag(key: String, value: Variant) -> void:
    _flags[key] = value
    EventManager.emit("flag_changed", {"key": key, "value": value})

func get_flag(key: String, default: Variant = false) -> Variant:
    return _flags.get(key, default)

func set_variable(key: String, value: Variant) -> void:
    _variables[key] = value

func get_variable(key: String, default: Variant = 0) -> Variant:
    return _variables.get(key, default)

func modify_variable(key: String, delta: float) -> void:
    var current = get_variable(key, 0)
    set_variable(key, current + delta)

func add_item(item_id: String, count: int = 1) -> void:
    for i in count:
        _items.append(item_id)

func remove_item(item_id: String, count: int = 1) -> bool:
    var removed = 0
    for i in range(_items.size() - 1, -1, -1):
        if _items[i] == item_id and removed < count:
            _items.remove_at(i)
            removed += 1
    return removed == count

func has_item(item_id: String) -> bool:
    return item_id in _items

func get_relationship(npc_id: String) -> int:
    return _relationships.get(npc_id, INITIAL_TRUST)

func modify_relationship(npc_id: String, delta: int) -> void:
    var current = get_relationship(npc_id)
    var new_value = clamp(current + delta, 0, 100)
    _relationships[npc_id] = new_value

func record_choice(choice_id: String) -> void:
    _choice_history.append({
        "choice_id": choice_id,
        "timestamp": Time.get_unix_time_from_system()
    })
```

```gdscript
## 网络状态
var _bandwidth: float = 0.0        # 当前带宽(Mbps)
var _latency: float = 0.0          # 当前延迟(ms)
var _current_quality: String = "720p"

## 阈值配置
const QUALITY_THRESHOLDS: Dictionary = {
    "1080p": {"min_bandwidth": 5.0, "max_latency": 200},
    "720p": {"min_bandwidth": 2.0, "max_latency": 300},
    "480p": {"min_bandwidth": 0.8, "max_latency": 500}
}

func _ready():
    # 启动网络监测
    _start_network_monitor()

func _start_network_monitor() -> void:
    var timer = Timer.new()
    timer.wait_time = 5.0
    timer.timeout.connect(_measure_network)
    add_child(timer)
    timer.start()

func _measure_network() -> void:
    # 使用Navigation Timing API测量带宽
    var bandwidth = JavaScriptBridge.eval("""
        (function() {
            var connection = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
            if (connection && connection.downlink) {
                return connection.downlink;
            }
            return 5.0; // 默认值
        })();
    """)
    _bandwidth = float(bandwidth)
    _update_quality()

func _update_quality() -> void:
    var new_quality = _calculate_best_quality()
    if new_quality != _current_quality:
        _current_quality = new_quality
        VideoManager.set_quality(new_quality)

func _calculate_best_quality() -> String:
    for quality in ["1080p", "720p", "480p"]:
        var threshold = QUALITY_THRESHOLDS[quality]
        if _bandwidth >= threshold["min_bandwidth"]:
            return quality
    return "480p"
```

```gdscript
var _preload_queue: Array = []          # 预加载队列

func _on_video_progress(video_id: String, progress: float) -> void:
    if progress >= PRELOAD_THRESHOLD:
        _start_preload()

func _start_preload() -> void:
    # 获取可能的下一节点
    var possible_nodes = StoryEngine.get_possible_next_nodes()
    var videos_to_preload = []

    for node_id in possible_nodes:
        var video_id = StoryEngine.get_video_id(node_id)
        if not _preloaded.has(video_id) and videos_to_preload.size() < MAX_PRELOAD:
            videos_to_preload.append(video_id)

    # 按优先级预加载
    for video_id in videos_to_preload:
        VideoManager.load_video(video_id, AdaptiveBitrateManager.get_current_quality())
        _preloaded[video_id] = true
```

```javascript
// 服务端签名生成
const crypto = require('crypto');

function generateSignedUrl(videoId, userId) {
    const expiresIn = 3600; // 1小时有效
    const expires = Math.floor(Date.now() / 1000) + expiresIn;

    const payload = `${videoId}|${userId}|${expires}`;
    const signature = crypto
        .createHmac('sha256', process.env.SIGN_SECRET)
        .update(payload)
        .digest('hex');

    const baseUrl = process.env.CDN_BASE_URL;
    return `${baseUrl}/videos/${videoId}.mp4?expires=${expires}&signature=${signature}`;
}
```

```gdscript
## 存档加密工具
class_name SaveEncryption

const KEY: String = "your-secret-key-here"  # 实际项目中应从安全配置读取

static func encrypt(data: String) -> String:
    var key_hash = KEY.sha256_text()
    var encrypted = ""

    for i in range(data.length()):
        var key_char = key_hash[i % key_hash.length()]
        var data_char = data[i]
        encrypted += char(ord(data_char) ^ ord(key_char))

    return Marshalls.utf8_to_base64(encrypted)

static func decrypt(encrypted_data: String) -> String:
    var data = Marshalls.base64_to_utf8(encrypted_data)
    var key_hash = KEY.sha256_text()
    var decrypted = ""

    for i in range(data.length()):
        var key_char = key_hash[i % key_hash.length()]
        var data_char = data[i]
        decrypted += char(ord(data_char) ^ ord(key_char))

    return decrypted
```
